Splay

变量

  1. ch[N][2]: 左右儿子.
  2. var[N]: x存储的值
  3. cnt[N]: 重复节点的个数。
  4. par[N]: x的父节点
  5. sz[N]: x子树下存储的节点个数

chk

查询节点和父节点之间的关系

int chk(int x){return ch[par[x]][1] == x;}//左关系返回0，右关系返回1.

pushup

更新sz数组

il void pushup(int x){sz[x] = sz[ch[x][0]] + sz[ch[x][1]] + cnt[x];}

rotate

il void rotate(int x){
    int y = par[x], z = par[y], d = chk(x), k = ch[x][d ^ 1];
    ch[y][d] = k, par[k] = y;
    ch[z][chk(y)] = x, par[x] = z;
    ch[x][d ^ 1] = y, par[y] = x;
    pushup(y), pushup(x);
}

splay

将一个节点旋转到指定节点的儿子，可以模拟下自下而上6个链状节点将3号节点旋转至根的情形。

void splay(int x, int goal = 0){
    while(par[x] != goal){
        int y = par[x], z = par[y];
        if(z != goal){
            if(chk(x) == chk(y)) rotate(y);
            else rotate(x);
        }
        rotate(x);
    }
    if(!goal) root = x;
}

find

将最大的小于等于x的节点伸展到根

void find(int x){
    int cur = root;
    while(ch[cur][x > val[cur]] && x != val[cur]) cur = ch[cur][x > val[cur]];
    splay(cur);
}

insert

记录父节点同时从根节点向下搜索。如果节点存在 cnt++, 否则新建节点并与父节点连边。

void insert(int x){
    int cur = root, p = 0;
    while(val[cur] != x && cur){
        p = cur;
        cur = ch[cur][x > val[cur]];
    }
    if(cur) cnt[cur]++;
    else {
        cur = ++ncnt;
        if(p) ch[p][x > val[p]] = cur;
        ch[cur][0] = ch[cur][1] = 0;
        par[cur] = p;val[cur] = x;
        sz[cur] = cnt[cur] = 1;
    }
    splay(cur);
}

Remark : 如果加入了哨兵节点请修改Kth 和 rank

kth

查询第k大的值

int kth(int k){
    int cur = root;
    while(1){
        if(k <= sz[ch[cur][0]] && ch[cur][0]) cur = ch[cur][0];
        else if(k > sz[ch[cur][0]] + cnt[cur]){
            k -= sz[ch[cur][0]] + cnt[cur];//注意顺序
            cur = ch[cur][1];
        }else{
            splay(cur);
            return cur;
        }
    }
}

rank

将节点find到根后返回左子树的大小即可（再加入哨兵节点的前提下）。

pre

将节点find到根后返回左子树最右边的节点即可

int pre(int x){
    find(x);
    if(val[root] < x) return root;
    int cur = ch[root][0];
    while(ch[cur][1]) cur = ch[cur][1];
    splay(cur);
    return cur;
}

suc

同pre

int suc(int x){
    find(x);
    if(val[root] > x) return root; 
    int cur = ch[root][1];
    while(ch[cur][0]) cur = ch[cur][0];
    splay(cur);
    return cur;
}

remove

把前驱伸展到根，后继伸展到根的右儿子，根的右儿子的左儿子就是要删除的节点

void remove(int x){
    int last = pre(x), nxt = suc(x);
    splay(last);splay(nxt, last);
    int del = ch[nxt][0];
    if(cnt[del] > 1){
        cnt[del]--;
        splay(del);
    }else ch[nxt][0] = 0;
    pushup(nxt), pushup(root);
}
