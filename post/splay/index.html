int chk(int x){return ch[par[x]][1] == x;}//左关系返回0，右关系返回1.

pushup

il void pushup(int x){sz[x] = sz[ch[x][0]] + sz[ch[x][1]] + cnt[x];}

rotate

void rotate(int x){
	int y = par[x], z = par[y], d = chk(x), k = ch[x][d ^ 1];
	ch[y][d] = k, par[k] = y;
	ch[z][chk(y)] = x, par[x] = z;
	ch[x][d ^ 1] = y, par[y] = x;
	pushup(y), pushup(x);
}

splay

将一个节点旋转到指定节点的儿子，可以模拟下自下而上6个链状节点将3号节点旋转至根的情形。

void splay(int x, int goal = 0){
	while(par[x] != goal){
		int y = par[x], z = par[y];
		if(z != goal){
			if(chk(x) == chk(y)) rotate(y);
			else rotate(x);
		}
		rotate(x);
	}
	if(!goal) root = x;
}

find

将最大的小于等于x的节点伸展到根

void find(int x){
	if(!root) return;
	int cur = root;
	while(ch[cur][x > cur] && val[cur] != x) cur = ch[cur][x > cur];
	splay(cur);
}
