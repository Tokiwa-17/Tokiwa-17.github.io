Lecture 9 :Graph

1.基本术语

  • 简单路径：路径上各顶点V1,V2,...,VmV_1, V_2, ..., V_mV1​,V2​,...,Vm​均不相互重复。
  • 连通图与强连通分量(无向图)
    • 连通图：任一对顶点都是联通的
    • 连通分量：非连通图的极大联通子图
  • 强联通图与强联通分量(有向图)
    • 强连通图：每一对顶点Vi,VjV_i, V_jVi​,Vj​,都可相互到达
    • 强连通分量：非强连通图的极大强连通子图。

virtual int firstNbr(int i){return nextNbr(i, n);} //首个邻接顶点
virtual int nextNbr(int i, int j){//相对于顶点 i 的下一邻接点
    while((j > -1) && !exist(i, --j)) ;//逆向线性试探
    return j;
}
  1. 边的基本操作
virtual bool exist(int i, int j){
return (i > -1) && (i < n) && (j > -1) && (j < n) && E[i][j] != NULL;
}

图的遍历

  • 遍历中组成树的一类边叫树边，与相应顶点构成遍历树。
  • 强调对特定状态顶点的查找，故称图搜索。
  • BFS、DFS, 均可在O(n + e)时间内完成。

//邻接表版本prim
struct Node{
    int v, dis;
};

vector<Node> Adj[MAXV];//Adj[u]存放u为始点连接的所有终点

int prim(){
    fill(d + 1, d + MAXV + 1, INF);
    d[1] = 0;
    int res = 0;
    for(int i = 0;i < n;i++){
		int u = -1, MIN = INF;
        for(int j = 1;j <= n;j++){
			if(!vis[j] && d[j] < MIN){
                u = j;
                MIN = d[j];
            }
        }
        if(u == -1) return -1;
        vis[u] = true;
        res += d[u];
        for(int j = 0;j < Adj[u].size();j++){
            int v = Adj[u][j].v;
            if(!vis[v] && Adj[u][v].dis < d[v])
                d[v] = Adj[u][v].dis;
        }
        return res;
    }
}

void minSpanTreePrim(MGraph g){
    int min, i, j, k;
    int pred[MAXVEX];//前驱
    int cutcost[MAXVEX];//到集合U的最小跨边大小
    //cutcost != 0意味着不在集合U中
    cutcost[0] = 0, pred[0] = 0;
    for(i = 1;i < g.numVertexes;i++){
        cutcost[i] = g.arc[0][i];
        pred[i] = 0;
    }
    for(i = 1;i < g.numVertexer;j++){//找离集合U最近的点
        min = INFINITY;
        for(j = 1;j < g.numVertexes; j++){
            if(cutcost[j] && cutcost[j] < min){
                min = cutcost[j];
                k = j;
            }
        }
        cout << pred[k] << ' ' << k;
        cutcost[k] = 0;
        for(j = 1; j < g.numVertexes;j++){//松弛操作
            if(cutcost[j] && g.arc[k][j] < cutcost[j]){
                cutcost[j] = g.arc[k][j];
                pred[j] = k;
            }
        }
    }
}
